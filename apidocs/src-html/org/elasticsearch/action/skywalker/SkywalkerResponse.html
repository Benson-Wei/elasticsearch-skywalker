019    package org.elasticsearch.action.skywalker;
020    
021    import java.io.IOException;
022    import java.util.HashMap;
023    import java.util.List;
024    import java.util.Map;
025    import org.elasticsearch.action.ShardOperationFailedException;
026    import org.elasticsearch.action.support.broadcast.BroadcastOperationResponse;
027    import org.elasticsearch.common.io.stream.StreamInput;
028    import org.elasticsearch.common.io.stream.StreamOutput;
029    
030    /**
031     * A response for a skywalker action.
032     *
033     */
034    public class SkywalkerResponse extends BroadcastOperationResponse {
035    
036        private Map<String, Map<String, Map<String, Object>>> response;
037    
038        SkywalkerResponse() {
039        }
040    
041        SkywalkerResponse(int totalShards, int successfulShards, int failedShards, List<ShardOperationFailedException> shardFailures) {
042            super(totalShards, successfulShards, failedShards, shardFailures);
043        }
044    
045        public SkywalkerResponse setResponse(Map<String, Map<String, Map<String, Object>>> response) {
046            this.response = response;
047            return this;
048        }
049    
050        public Map<String, Map<String, Map<String, Object>>> getResponse() {
051            return response;
052        }
053    
054        @Override
055        public void readFrom(StreamInput in) throws IOException {
056            super.readFrom(in);
057            int indexCount = in.readInt();
058            this.response = new HashMap();
059            for (int i = 0; i < indexCount; i++) {
060                String index = in.readUTF();
061                Map<String, Map<String, Object>> shards = new HashMap();
062                int shardCount = in.readInt();
063                for (int j = 0; j < shardCount; j++) {
064                    String shard = in.readUTF();
065                    Map<String, Object> fields = in.readMap();
066                    shards.put(shard, fields);
067                }
068                response.put(index, shards);
069            }
070        }
071    
072        @Override
073        public void writeTo(StreamOutput out) throws IOException {
074            super.writeTo(out);
075            int indexCount = response.keySet().size(); 
076            out.writeInt(indexCount);
077            for (String index : response.keySet()) {
078                out.writeUTF(index);
079                int shardCount = response.get(index).keySet().size();
080                out.writeInt(shardCount);
081                for (String shard : response.get(index).keySet()) {
082                    out.writeUTF(shard);
083                    out.writeMap(response.get(index).get(shard));
084                }
085            }
086        }
087    }
