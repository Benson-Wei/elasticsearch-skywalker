105        @Override
106        protected SkywalkerResponse newResponse(SkywalkerRequest request, AtomicReferenceArray shardsResponses, ClusterState clusterState) {
107            int successfulShards = 0;
108            int failedShards = 0;
109            List<ShardOperationFailedException> shardFailures = null;
110            Map<String, Map<String, Map<String, Object>>> response = new HashMap();
111            for (int i = 0; i < shardsResponses.length(); i++) {
112                Object shardResponse = shardsResponses.get(i);
113                if (shardResponse == null) {
114                    // a non active shard, ignore...
115                } else if (shardResponse instanceof BroadcastShardOperationFailedException) {
116                    failedShards++;
117                    if (shardFailures == null) {
118                        shardFailures = newArrayList();
119                    }
120                    shardFailures.add(new DefaultShardOperationFailedException((BroadcastShardOperationFailedException) shardResponse));
121                } else {
122                    successfulShards++;
123                    if (shardResponse instanceof ShardSkywalkerResponse) {
124                        ShardSkywalkerResponse shardResp = (ShardSkywalkerResponse) shardResponse;
125                        String index = shardResp.getIndex();
126                        int shardId = shardResp.shardId();
127                        // one map per index
128                        Map indexresponse = response.get(index);
129                        if (indexresponse == null) {
130                            indexresponse = new HashMap();
131                        }
132                        // merge index-wide fieldInfo into single field
133                        indexresponse.put("fieldInfos", shardResp.getResponse().get("fieldInfos"));
134                        shardResp.getResponse().remove("fieldInfos");
135                        // shard-wise data
136                        indexresponse.put(Integer.toString(shardId), shardResp.getResponse());
137                        response.put(index, indexresponse);
138                    }
139                }
140            }
141            return new SkywalkerResponse(shardsResponses.length(), successfulShards, failedShards, shardFailures).setResponse(response);
142        }

159        @Override
160        protected ShardSkywalkerResponse shardOperation(ShardSkywalkerRequest request) throws ElasticSearchException {
161            synchronized (mutex) {
162                try {
163                    IndexService indexService = indicesService.indexServiceSafe(request.index());
164                    InternalIndexShard indexShard = (InternalIndexShard) indexService.shardSafe(request.shardId());
165                    Map<String, Object> response = new HashMap();
166                    Store store = indexShard.store();
167                    Directory directory = store.directory();
168                    IndexReader reader = indexShard.searcher().reader();
169                    IndexInfo indexInfo = new IndexInfo(reader);
170                    getStoreMetadata(response, store.list());
171                    response.put("numDocs", reader.numDocs());
172                    response.put("maxDoc", reader.maxDoc());
173                    response.put("numDeletedDocs", reader.numDeletedDocs());
174                    response.put("numTerms", indexInfo.getNumTerms());
175                    response.put("hasDeletions", reader.hasDeletions());
176                    response.put("indexversion", indexInfo.getVersion());
177                    Map indexFormat = new HashMap();
178                    indexFormat.put("id", indexInfo.getIndexFormat());
179                    indexFormat.put("genericName", indexInfo.getFormatDetails().genericName);
180                    indexFormat.put("capabilities", indexInfo.getFormatDetails().capabilities);
181                    response.put("indexFormat", indexFormat);
182                    response.put("directoryImpl", indexInfo.getDirImpl());
183                    List commits = new ArrayList();
184                    for (IndexCommit ic : reader.listCommits(directory)) {
185                        Map commitMap = new HashMap();
186                        commitMap.put("segment", ic.getSegmentsFileName());
187                        commitMap.put("count", ic.getSegmentCount());
188                        commitMap.put("deleted", ic.isDeleted());
189                        commitMap.put("files", ic.getFileNames());
190                        commitMap.put("userdata", ic.getUserData());
191                        commits.add(commitMap);
192                    }
193                    response.put("commits", commits);
194                    List fieldInfos = new ArrayList();
195                    Map<String, Type> typeMap = new HashMap();
196                    for (FieldInfo fi : ReaderUtil.getMergedFieldInfos(reader)) {
197                        fieldInfos.add(getFieldInfo(indexService, fi, typeMap));
198                    }
199                    response.put("fieldInfos", fieldInfos);
200                    List termList = new ArrayList();
201                    for (TermInfo ti : indexInfo.getTopTerms()) {
202                        Map tiMap = new HashMap();
203                        tiMap.put("field", ti.term.field());
204                        tiMap.put("text", decode(ti.term, typeMap));
205                        tiMap.put("docFreq", ti.docFreq);
206                        termList.add(tiMap);
207                    }
208                    response.put("topterms", termList);
209                    return new ShardSkywalkerResponse(request.index(), request.shardId()).setResponse(response);
210                } catch (IOException ex) {
211                    throw new ElasticSearchException(ex.getMessage(), ex);
212                }
213            }
214        }

231        private Map<String, Object> getFieldInfo(IndexService indexService, FieldInfo fieldinfo, Map<String, Type> types) {
232            Map<String, Object> info = new HashMap();
233            info.put("name", fieldinfo.name);
234            info.put("isindexed", fieldinfo.isIndexed);
235            info.put("storePayloads", fieldinfo.storePayloads);
236            info.put("storeTermVector", fieldinfo.storeTermVector);
237            info.put("number", fieldinfo.number);
238            info.put("omitNorms", fieldinfo.omitNorms);
239            info.put("options", fieldinfo.indexOptions.name().toString());
240            MapperService mapperService = indexService.mapperService();
241            if (mapperService == null) {
242                return info;
243            }
244            FieldMapper fieldMapper = mapperService.smartNameFieldMapper(fieldinfo.name);
245            if (fieldMapper != null) {
246                Map<String, Object> mapper = new HashMap();
247                mapper.put("analyzed", fieldMapper.analyzed());
248                mapper.put("boost", fieldMapper.boost());
249                mapper.put("indexed", fieldMapper.indexed());
250                mapper.put("omitNorms", fieldMapper.omitNorms());
251                mapper.put("omitTermFreqAndPositions", fieldMapper.omitTermFreqAndPositions());
252                Type type = FieldType.type(fieldMapper.fieldDataType());
253                mapper.put("fieldDataType", type.toString());
254                types.put(fieldinfo.name, type);
255                mapper.put("fullName", fieldMapper.names().fullName());
256                mapper.put("indexName", fieldMapper.names().indexName());
257                mapper.put("indexNameClean", fieldMapper.names().indexNameClean());
258                info.put("mapper", mapper);
259            }
260            return info;
261        }

263        private void getStoreMetadata(Map<String, Object> response, ImmutableMap<String, StoreFileMetaData> metadata) {
264            List result = new ArrayList();
265            long minLastModified = Long.MAX_VALUE;
266            long maxLastModified = Long.MIN_VALUE;
267            for (String name : metadata.keySet()) {
268                StoreFileMetaData m = metadata.get(name);
269                Map<String, Object> info = new HashMap();
270                info.put("name", name);
271                info.put("length", m.length());
272                if (m.lastModified() < minLastModified) {
273                    minLastModified = m.lastModified();
274                }
275                if (m.lastModified() > maxLastModified) {
276                    maxLastModified = m.lastModified();
277                }
278                info.put("lastmodified", DATE_TIME_FORMATTER.printer().print(m.lastModified()));
279                info.put("checksum", m.checksum());
280                info.put("func", IndexGate.getFileFunction(name));
281                result.add(info);
282            }
283            response.put("store", result);
284            response.put("minlastmodified", minLastModified);
285            response.put("maxlastmodified", maxLastModified);
286        }

288        private String decode(Term term, Map<String, Type> typeMap) {
289            String value = term.text();
290            Type type = typeMap.get(term.field());
291            switch (type) {
292                case STRING:
293                    return term.text();
294                case BYTE:
295                    byte[] b = value.getBytes();
296                    return bytesToHex(b, 0, b.length);
297                case INT:
298                case SHORT:
299                    return Integer.toString(NumericUtils.prefixCodedToInt(value));
300                case LONG:
301                    return Long.toString(NumericUtils.prefixCodedToLong(value));
302                case FLOAT:
303                    return Float.toString(NumericUtils.prefixCodedToFloat(value));
304                case DOUBLE:
305                    return Double.toString(NumericUtils.prefixCodedToDouble(value));
306                case NULL:
307                    return "null";
308    
309            }
310            return "unknown";
311        }
312    
313        private String bytesToHex(byte bytes[], int offset, int length) {
314            StringBuilder sb = new StringBuilder();
315            for (int i = offset; i < offset + length; ++i) {
316                if (i > offset) {
317                    sb.append(" ");
318                }
319                sb.append(Integer.toHexString(0x0100 + (bytes[i] & 0x00FF)).substring(1));
320            }
321            return sb.toString();
322        }
