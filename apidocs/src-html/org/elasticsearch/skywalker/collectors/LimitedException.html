019    package org.elasticsearch.skywalker.collectors;
020    
021    public class LimitedException extends RuntimeException {
022    
023        private int limitType;
024        private long limitSize;
025        private int lastDoc;
026        private long currentSize;
027    
028        public LimitedException(int limitType, long limitSize, long currentSize, int lastDoc) {
029            super();
030            this.limitType = limitType;
031            this.limitSize = limitSize;
032            this.currentSize = currentSize;
033            this.lastDoc = lastDoc;
034        }
035    
036        /**
037         * @return the limitType
038         */
039        public int getLimitType() {
040            return limitType;
041        }
042    
043        /**
044         * @param limitType the limitType to set
045         */
046        public void setLimitType(int limitType) {
047            this.limitType = limitType;
048        }
049    
050        /**
051         * @return the limitSize
052         */
053        public long getLimitSize() {
054            return limitSize;
055        }
056    
057        /**
058         * @param limitSize the limitSize to set
059         */
060        public void setLimitSize(long limitSize) {
061            this.limitSize = limitSize;
062        }
063    
064        /**
065         * @return the currentSize
066         */
067        public long getCurrentSize() {
068            return currentSize;
069        }
070    
071        /**
072         * @param currentSize the currentSize to set
073         */
074        public void setCurrentSize(long currentSize) {
075            this.currentSize = currentSize;
076        }
077    
078        /**
079         * @return the lastDoc
080         */
081        public int getLastDoc() {
082            return lastDoc;
083        }
084    
085        /**
086         * @param lastDoc the lastDoc to set
087         */
088        public void setLastDoc(int lastDoc) {
089            this.lastDoc = lastDoc;
090        }
091    }
