019    package org.elasticsearch.skywalker;
020    
021    import java.io.IOException;
022    import java.util.ArrayList;
023    import java.util.Collections;
024    import java.util.Iterator;
025    import java.util.List;
026    import org.apache.lucene.index.FieldInfo;
027    import org.apache.lucene.index.FormatDetails;
028    import org.apache.lucene.index.IndexGate;
029    import org.apache.lucene.index.IndexReader;
030    import org.apache.lucene.index.TermEnum;
031    import org.apache.lucene.store.Directory;
032    import org.apache.lucene.util.ReaderUtil;
033    
034    public class IndexInfo {
035    
036        private IndexReader reader;
037        private Directory dir;
038        private int numTerms;
039        private int indexFormat;
040        private FormatDetails formatDetails;
041        private TermInfo[] topTerms;
042        private List<FieldInfo> fieldInfo;
043        private List<String> fieldNames;
044        private String version;
045        private String dirImpl;
046    
047        public IndexInfo(IndexReader reader) throws IOException {
048            this.reader = reader;
049            try {
050                this.dir = reader.directory();
051                this.dirImpl = dir.getClass().getName();
052            } catch (UnsupportedOperationException uoe) {
053                this.dir = null;
054                this.dirImpl = "N/A";
055            }
056            try {
057                this.version = Long.toString(reader.getVersion());
058            } catch (UnsupportedOperationException uoe) {
059                this.dir = null;
060            }
061            numTerms = 0;
062            TermEnum te = null;
063            try {
064                te = reader.terms();
065                while (te.next()) {
066                    numTerms++;
067                }
068            } finally {
069                te.close();
070            }
071            fieldInfo = new ArrayList();
072            fieldNames = new ArrayList();        
073            final Iterator<FieldInfo> it = ReaderUtil.getMergedFieldInfos(reader).iterator();
074            while (it.hasNext()) {
075                final FieldInfo info = it.next();            
076                fieldInfo.add(info);
077                fieldNames.add(info.name);
078            }
079            Collections.sort(fieldNames);
080            if (dir != null) {
081                indexFormat = IndexGate.getIndexFormat(dir);
082                formatDetails = IndexGate.getFormatDetails(indexFormat);
083            } else {
084                indexFormat = -1;
085                formatDetails = new FormatDetails();
086            }
087            topTerms = HighFreqTerms.getHighFreqTerms(reader, null, 51, (String[]) fieldNames.toArray(new String[fieldNames.size()]));
088        }
089    
090        /**
091         * @return the reader
092         */
093        public IndexReader getReader() {
094            return reader;
095        }
096    
097        public Directory getDirectory() {
098            return dir;
099        }
100    
101        /**
102         * @return the numTerms
103         */
104        public int getNumTerms() {
105            return numTerms;
106        }
107    
108        /**
109         * @return the indexFormat
110         */
111        public int getIndexFormat() {
112            return indexFormat;
113        }
114    
115        /**
116         * @return the formatDetails
117         */
118        public FormatDetails getFormatDetails() {
119            return formatDetails;
120        }
121    
122        /**
123         * @return the topTerms
124         */
125        public TermInfo[] getTopTerms() {
126            return topTerms;
127        }
128    
129        public List<FieldInfo> getFieldInfos() {
130            return fieldInfo;
131        }
132        
133        /**
134         * @return the fieldNames
135         */
136        public List<String> getFieldNames() {
137            return fieldNames;
138        }
139    
140        public String getVersion() {
141            return version;
142        }
143    
144        public String getDirImpl() {
145            return dirImpl;
146        }
147    }
