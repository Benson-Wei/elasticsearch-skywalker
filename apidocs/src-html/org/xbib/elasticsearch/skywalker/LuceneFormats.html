019package org.xbib.elasticsearch.skywalker;
020
021/**
022 * Lucene format constants
023 *
024 *  @author <a href="mailto:joergprante@gmail.com">J&ouml;rg Prante</a>
025 */
026public interface LuceneFormats {
027
028    // old version constants
029    int OLD_FORMAT = -1;
030
031    /**
032     * This format adds details used for lockless commits.  It differs
033     * slightly from the previous format in that file names
034     * are never re-used (write once).  Instead, each file is
035     * written to the next generation.  For example,
036     * segments_1, segments_2, etc.  This allows us to not use
037     * a commit lock.  See <a
038     * href="http://lucene.apache.org/java/docs/fileformats.html">file
039     * formats</a> for details.
040     */
041    int FORMAT_LOCKLESS = -2;
042
043    /**
044     * This format adds a "hasSingleNormFile" flag into each segment info.
045     * See <a href="http://issues.apache.org/jira/browse/LUCENE-756">LUCENE-756</a>
046     * for details.
047     */
048    int FORMAT_SINGLE_NORM_FILE = -3;
049
050    /**
051     * This format allows multiple segments to share a single
052     * vectors and stored fields file.
053     */
054    int FORMAT_SHARED_DOC_STORE = -4;
055
056    /**
057     * This format adds a checksum at the end of the file to
058     * ensure all bytes were successfully written.
059     */
060    int FORMAT_CHECKSUM = -5;
061
062    /**
063     * This format adds the deletion count for each segment.
064     * This way IndexWriter can efficiently report numDocs().
065     */
066    int FORMAT_DEL_COUNT = -6;
067
068    /**
069     * This format adds the boolean hasProx to record if any
070     * fields in the segment store prox information (ie, have
071     * omitTermFreqAndPositions==false)
072     */
073    int FORMAT_HAS_PROX = -7;
074
075    /**
076     * This format adds optional commit userData (String) storage.
077     */
078    int FORMAT_USER_DATA = -8;
079
080    /**
081     * This format adds optional per-segment String
082     * diagnostics storage, and switches userData to Map
083     */
084    int FORMAT_DIAGNOSTICS = -9;
085
086    /**
087     * Each segment records whether it has term vectors
088     */
089    int FORMAT_HAS_VECTORS = -10;
090
091    /**
092     * Each segment records the Lucene version that created it.
093     */
094    int FORMAT_3_1 = -11;
095    /**
096     * Some early 4.0 pre-alpha
097     */
098    int FORMAT_PRE_4 = -12;
099}
