130        public String getFileFunction(String file) {
131            if (file == null || file.trim().length() == 0) {
132                return file;
133            }
134            String res = "undefined";
135            file = file.trim();
136            int idx = file.indexOf('.');
137            String suffix = null;
138            if (idx != -1) {
139                suffix = file.substring(idx + 1);
140            }
141            if (suffix != null) {
142                res = knownExtensions.get(suffix);
143                if (res != null) {
144                    return res;
145                }
146                // perhaps per-field norms?
147                if (suffix.length() == 2) {
148                    return knownExtensions.get(suffix.substring(0, 1));
149                }
150            }
151            if (file.startsWith("segments_")) {
152                return "Lucene segment";
153            }
154            // elasticsearch checksums
155            if (file.startsWith("_checksum")) {
156                return "Elasticsearch checksum file";
157            }
158            return res;
159        }

161        private FormatDetails detectOldFormats(int format) {
162            switch (format) {
163                case OLD_FORMAT:
164                    return new FormatDetails("old plain", "Lucene Pre-2.1", "2.0?");
165                case FORMAT_LOCKLESS:
166                    return new FormatDetails("lock-less", "Lucene 2.1", "2.1");
167                case FORMAT_SINGLE_NORM_FILE:
168                    return new FormatDetails("lock-less, single norms file", "Lucene 2.2", "2.2");
169                case FORMAT_SHARED_DOC_STORE:
170                    return new FormatDetails("lock-less, single norms file, shared doc store", "Lucene 2.3", "2.3");
171                case FORMAT_CHECKSUM:
172                    return new FormatDetails("lock-less, single norms, shared doc store, checksum", "Lucene 2.4", "2.4");
173                case FORMAT_DEL_COUNT:
174                    return new FormatDetails("lock-less, single norms, shared doc store, checksum, del count", "Lucene 2.4", "2.4");
175                case FORMAT_HAS_PROX:
176                    return new FormatDetails("lock-less, single norms, shared doc store, checksum, del count, omitTf", "Lucene 2.4", "2.4");
177                case FORMAT_USER_DATA:
178                    return new FormatDetails("lock-less, single norms, shared doc store, checksum, del count, omitTf, user data", "Lucene 2.9-dev", "2.9-dev");
179                case FORMAT_DIAGNOSTICS:
180                    return new FormatDetails("lock-less, single norms, shared doc store, checksum, del count, omitTf, user data, diagnostics", "Lucene 2.9", "2.9");
181                case FORMAT_HAS_VECTORS:
182                    return new FormatDetails("lock-less, single norms, shared doc store, checksum, del count, omitTf, user data, diagnostics, hasVectors", "Lucene 2.9", "2.9");
183                case FORMAT_3_1:
184                    return new FormatDetails("lock-less, single norms, shared doc store, checksum, del count, omitTf, user data, diagnostics, hasVectors", "Lucene 3.1", "3.1");
185                case FORMAT_PRE_4:
186                    return new FormatDetails("flexible, unreleased 4.0 pre-alpha", "Lucene 4.0-dev", "4.0-dev");
187                default:
188                    if (format < FORMAT_PRE_4) {
189                        return new FormatDetails("flexible, unreleased 4.0 pre-alpha", "Lucene 4.0-dev", "4.0-dev");
190                    } else {
191                        return new FormatDetails("unknown", "Lucene 1.3 or earlier, or unreleased", "1.3?");
192                    }
193            }
194        }

196        public FormatDetails getIndexFormat(final Directory dir) {
197            SegmentInfos.FindSegmentsFile fsf = new SegmentInfos.FindSegmentsFile(dir) {
198    
199                protected Object doBody(String segmentsFile) throws IOException {
200                    FormatDetails res = new FormatDetails("unknown", "unknown", "-1");
201                    IndexInput in = dir.openInput(segmentsFile, IOContext.READ);
202                    try {
203                        int indexFormat = in.readInt();
204                        if (indexFormat == CodecUtil.CODEC_MAGIC) {
205                            res.setCapabilities("flexible, codec-specific");
206                            res.setGenericName("Lucene 4.x");
207                            int actualVersion = SegmentInfos.VERSION_40;
208                            try {
209                                actualVersion = CodecUtil.checkHeaderNoMagic(in, "segments", SegmentInfos.VERSION_40, Integer.MAX_VALUE);
210                                if (actualVersion > SegmentInfos.VERSION_40) {
211                                    res.setCapabilities(res.getCapabilities() + " (WARNING: newer version of Lucene than this tool)");
212                                }
213                            } catch (Exception e) {
214                                res.setCapabilities(res.getCapabilities() +
215                                        " (error reading: " + e.getMessage() + ")");
216                            }
217                            res.setGenericName("Lucene 4." + actualVersion);
218                            res.setVersion("4." + actualVersion);
219                        } else {
220                            res = detectOldFormats(indexFormat);
221                            res.setGenericName(res.getGenericName() + " (" + indexFormat + ")");
222                            if (res.getVersion().compareTo("3") < 0) {
223                                res.setCapabilities(res.getCapabilities() + " (UNSUPPORTED)");
224                            }
225                        }
226                    } finally {
227                        in.close();
228                    }
229                    return res;
230                }
231            };
232            try {
233                return (FormatDetails) fsf.run();
234            } catch (IOException e) {
235                return null;
236            }
237        }

246        public List<String> getIndexFiles(Directory dir) {
247            List<IndexCommit> commits;
248            try {
249                commits = DirectoryReader.listCommits(dir);
250            } catch (IndexNotFoundException e) {
251                return Collections.emptyList();
252            } catch (IOException e) {
253                return Collections.emptyList();
254            }
255            Set<String> known = new HashSet<String>();
256            try {
257                for (IndexCommit ic : commits) {
258                    known.addAll(ic.getFileNames());
259                }
260                if (dir.fileExists(IndexFileNames.SEGMENTS_GEN)) {
261                    known.add(IndexFileNames.SEGMENTS_GEN);
262                }
263            } catch (IOException e) {
264                // ignore
265            }
266            return new ArrayList<String>(known);
267        }

270            long totalFileSize = 0L;
271            String[] files;
272            files = directory.listAll();
273            if (files == null) {
274                return totalFileSize;
275            }
276            for (String file : files) {
277                totalFileSize += directory.fileLength(file);
278            }
279            return totalFileSize;
280        }

283            Map<String, Object> m = new HashMap();
284            m.put("name", fi.name);
285            m.put("number", fi.number);
286            m.put("isIndexed", fi.isIndexed());
287            m.put("hasDocValues", fi.hasDocValues());
288            m.put("hasNorms", fi.hasNorms());
289            m.put("hasPayloads", fi.hasPayloads());
290            m.put("hasVectors", fi.hasVectors());
291            if (fi.getDocValuesType() != null) {
292                m.put("docValuesType", fi.getDocValuesType().name());
293            }
294            if (fi.getNormType() != null) {
295                m.put("normType", fi.getNormType().name());
296            }
297            if (fi.getIndexOptions() != null) {
298                m.put("options", fi.getIndexOptions().name());
299            }
300            m.put("attributes", fi.attributes());
301            FieldMapper fieldMapper = mapperService.smartNameFieldMapper(fi.name);
302            if (fieldMapper != null) {
303                Map<String, Object> mapper = new HashMap();
304                mapper.put("fullName", fieldMapper.names().fullName());
305                mapper.put("indexName", fieldMapper.names().indexName());
306                mapper.put("indexNameClean", fieldMapper.names().indexNameClean());
307    
308                mapper.put("boost", fieldMapper.boost());
309    
310                if (fieldMapper.indexAnalyzer() != null) {
311                    mapper.put("indexAnalyzer", fieldMapper.indexAnalyzer().toString());
312                }
313                if (fieldMapper.searchAnalyzer() != null) {
314                    mapper.put("searchAnalyzer", fieldMapper.searchAnalyzer().toString());
315                }
316                if (fieldMapper.searchQuoteAnalyzer() != null) {
317                    mapper.put("searchQuoteAnalyzer", fieldMapper.searchQuoteAnalyzer().toString());
318                }
319    
320                FieldDataType dataType = fieldMapper.fieldDataType();
321                if (dataType != null) {
322                    mapper.put("fieldDataType", dataType.getType());
323                }
324    
325                FieldType type = fieldMapper.fieldType();
326                if (type != null) {
327                    mapper.put("indexed", type.indexed());
328                    mapper.put("stored", type.stored());
329                    mapper.put("tokenized", type.tokenized());
330                    mapper.put("omitNorms", type.omitNorms());
331                    mapper.put("storeTermVectors", type.storeTermVectors());
332                    mapper.put("storeTermVectorOffsets", type.storeTermVectorOffsets());
333                    mapper.put("storeTermVectorPayloads", type.storeTermVectorPayloads());
334                    mapper.put("storeTermVectorPositions", type.storeTermVectorPositions());
335                    if (type.numericType() != null) {
336                        mapper.put("numericType", type.numericType().name());
337                        mapper.put("numericPrecisionStep", type.numericPrecisionStep());
338                    }
339                    if (type.docValueType() != null) {
340                        mapper.put("docValueType", type.docValueType().name());
341                    }
342                }
343    
344                SimilarityProvider similarityProvider = fieldMapper.similarity();
345                if (similarityProvider != null) {
346                    mapper.put("similarityPovider", similarityProvider.name());
347                    mapper.put("similarity", similarityProvider.get().getClass().getName() );
348                }
349    
350                PostingsFormatProvider postingsFormatProvider = fieldMapper.postingsFormatProvider();
351                if (postingsFormatProvider != null) {
352                    mapper.put("postingsFormatProvider", postingsFormatProvider.name());
353                    mapper.put("postingsFormat", postingsFormatProvider.get().getName());
354                }
355    
356                m.put("mapper", mapper);
357            }
358            return m;
359        }

362            List<Map<String, Object>> result = new ArrayList();
363            for (String name : metadata.keySet()) {

382        public static MetaData.Builder loadState(List<File> files, NodeEnvironment nodeEnv) throws ElasticsearchException {
383            MetaData.Builder metaDataBuilder;
384            try {
385                MetaData globalMetaData = loadGlobalState(files, nodeEnv);
386                if (globalMetaData != null) {
387                    metaDataBuilder = MetaData.builder(globalMetaData);
388                } else {
389                    metaDataBuilder = MetaData.builder();
390                }
391                Set<String> indices = nodeEnv.findAllIndices();
392                for (String index : indices) {
393                    IndexMetaData indexMetaData = loadIndex(files, index, nodeEnv);
394                    if (indexMetaData == null) {
395                        continue;
396                    } else {
397                        metaDataBuilder.put(indexMetaData, false);
398                    }
399                }
400            } catch (Exception e) {
401                throw new ElasticsearchException(e.getMessage());
402            }
403            return metaDataBuilder;
404        }

406        @Nullable
407        private static IndexMetaData loadIndex(List<File> files, String index, NodeEnvironment nodeEnv) {
408            long highestVersion = -1;
409            IndexMetaData indexMetaData = null;
410            for (File indexLocation : nodeEnv.indexLocations(new Index(index))) {
411                File stateDir = new File(indexLocation, "_state");
412                if (!stateDir.exists() || !stateDir.isDirectory()) {
413                    continue;
414                }
415                // now, iterate over the current versions, and find latest one
416                File[] stateFiles = stateDir.listFiles();
417                if (stateFiles == null) {
418                    continue;
419                }
420                for (File stateFile : stateFiles) {
421                    if (!stateFile.getName().startsWith("state-")) {
422                        continue;
423                    }
424                    files.add(stateFile);
425                    try {
426                        long version = Long.parseLong(stateFile.getName().substring("state-".length()));
427                        if (version > highestVersion) {
428                            byte[] data = Streams.copyToByteArray(new FileInputStream(stateFile));
429                            if (data.length == 0) {
430                                continue;
431                            }
432                            XContentParser parser = null;
433                            try {
434                                parser = XContentHelper.createParser(data, 0, data.length);
435                                parser.nextToken(); // move to START_OBJECT
436                                indexMetaData = IndexMetaData.Builder.fromXContent(parser);
437                                highestVersion = version;
438                            } finally {
439                                if (parser != null) {
440                                    parser.close();
441                                }
442                            }
443                        }
444                    } catch (Exception e) {
445                        continue;
446                    }
447                }
448            }
449            return indexMetaData;
450        }

452        private static MetaData loadGlobalState(List<File> files, NodeEnvironment nodeEnv) {
453            long highestVersion = -1;
454            MetaData metaData = null;
455            for (File dataLocation : nodeEnv.nodeDataLocations()) {
456                File stateLocation = new File(dataLocation, "_state");
457                if (!stateLocation.exists()) {
458                    continue;
459                }
460                File[] stateFiles = stateLocation.listFiles();
461                if (stateFiles == null) {
462                    continue;
463                }
464                for (File stateFile : stateFiles) {
465                    if (!stateFile.getName().startsWith("global-")) {
466                        continue;
467                    }
468                    files.add(stateFile);
469                    try {
470                        long version = Long.parseLong(stateFile.getName().substring("global-".length()));
471                        if (version > highestVersion) {
472                            byte[] data = Streams.copyToByteArray(new FileInputStream(stateFile));
473                            if (data.length == 0) {
474                                continue;
475                            }
476                            XContentParser parser = null;
477                            try {
478                                parser = XContentHelper.createParser(data, 0, data.length);
479                                metaData = MetaData.Builder.fromXContent(parser);
480                                highestVersion = version;
481                            } finally {
482                                if (parser != null) {
483                                    parser.close();
484                                }
485                            }
486                        }
487                    } catch (Exception e) {
488                        continue;
489                    }
490                }
491            }
492            return metaData;
493        }

495        public Set<FieldTermCount> getFieldTermCounts() throws IOException {
496            Set<FieldTermCount> termCounts = new TreeSet<FieldTermCount>();
497            numTerms = 0;
498            Fields fields = MultiFields.getFields(reader);
499            if (fields != null) {
500                Iterator<String> fe = fields.iterator();
501                String fld;
502                TermsEnum te = null;
503                while (fe.hasNext()) {
504                    fld = fe.next();
505                    long termCount = 0L;
506                    Terms terms = fields.terms(fld);
507                    if (terms != null) {
508                        te = terms.iterator(te);
509                        while (te.next() != null) {
510                            termCount++;
511                            numTerms++;
512                        }
513                    }
514                    termCounts.add(new FieldTermCount(fld, termCount));
515                }
516            }
517            return termCounts;
518        }

531        private static final TermStats[] EMPTY_STATS = new TermStats[0];
532    
533        public TermStats[] getHighFreqTerms(int numTerms, String[] fieldNames) {
534            TermStatsQueue tiq = new TermStatsQueue(numTerms);
535            TermsEnum te = null;
536            try {
537                if (fieldNames != null) {
538                    Fields fields = MultiFields.getFields(reader);
539                    if (fields == null) {
540                        return EMPTY_STATS;
541                    }
542                    for (String field : fieldNames) {
543                        Terms terms = fields.terms(field);
544                        if (terms != null) {
545                            te = terms.iterator(te);
546                            fillQueue(te, tiq, field);
547                        }
548                    }
549                } else {
550                    Fields fields = MultiFields.getFields(reader);
551                    if (fields == null) {
552                        return EMPTY_STATS;
553                    }
554                    for (String field : fields) {
555                        Terms terms = fields.terms(field);
556                        te = terms.iterator(te);
557                        fillQueue(te, tiq, field);
558                    }
559                }
560            } catch (IOException e) {
561                // ignore
562            }
563            TermStats[] result = new TermStats[tiq.size()];
564            // we want highest first so we read the queue and populate the array
565            // starting at the end and work backwards
566            int count = tiq.size() - 1;
567            while (tiq.size() != 0) {
568                result[count] = tiq.pop();
569                count--;
570            }
571            return result;
572        }

575            return bytesToHex(bytes.bytes, bytes.offset, bytes.length, wrap);

578        public static String bytesToHex(byte bytes[], int offset, int length, boolean wrap) {
579            StringBuilder sb = new StringBuilder();
580            boolean newLine = false;
581            for (int i = offset; i < offset + length; ++i) {
582                if (i > offset && !newLine) {
583                    sb.append(" ");
584                }
585                sb.append(Integer.toHexString(0x0100 + (bytes[i] & 0x00FF))
586                        .substring(1));
587                if (i > 0 && (i + 1) % 16 == 0 && wrap) {
588                    sb.append("\n");
589                    newLine = true;
590                } else {
591                    newLine = false;
592                }
593            }
594            return sb.toString();
595        }

597        private void fillQueue(TermsEnum termsEnum, TermStatsQueue tiq, String field) {
598            while (true) {
599                try {
600                    BytesRef term = termsEnum.next();
601                    if (term != null) {
602                        BytesRef text = new BytesRef();
603                        text.copyBytes(term);
604                        TermStats ts = new TermStats();
605                        ts.field(field).text(text).docFreq(termsEnum.docFreq());
606                        tiq.insertWithOverflow(ts);
607                    } else {
608                        break;
609                    }
610                } catch (IOException e) {
611                    break;
612                }
613            }
614        }
