002package org.xbib.elasticsearch.skywalker.reconstruct;
003
004import java.util.Comparator;
005
006/**
007 *  Document term comparator
008 */
009public class DocumentTermComparator implements Comparator<DocumentTerm> {
010    private boolean ascending;
011    private boolean byText;
012
013    public DocumentTermComparator(boolean byText, boolean ascending) {
014        this.ascending = ascending;
015        this.byText = byText;
016    }
017
018    public int compare(DocumentTerm h1, DocumentTerm h2) {
019        if (byText) {
020            return ascending ? h1.text().compareTo(h2.text()) : h2.text().compareTo(h1.text());
021        } else {
022            if (h1.count() > h2.count()) {
023                return ascending ? -1 : 1;
024            }
025            if (h1.count() < h2.count()) {
026                return ascending ? 1 : -1;
027            }
028        }
029        return 0;
030    }
031}
