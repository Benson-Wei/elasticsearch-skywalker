001    package org.xbib.elasticsearch.skywalker.reconstruct;
002    
003    import org.apache.lucene.document.Document;
004    import org.apache.lucene.index.AtomicReader;
005    import org.apache.lucene.index.CompositeReader;
006    import org.apache.lucene.index.DocsAndPositionsEnum;
007    import org.apache.lucene.index.FieldInfo;
008    import org.apache.lucene.index.FieldInfos;
009    import org.apache.lucene.index.IndexReader;
010    import org.apache.lucene.index.IndexableField;
011    import org.apache.lucene.index.MultiFields;
012    import org.apache.lucene.index.SlowCompositeReaderWrapper;
013    import org.apache.lucene.index.Terms;
014    import org.apache.lucene.index.TermsEnum;
015    import org.apache.lucene.util.Bits;
016    import org.elasticsearch.ElasticsearchIllegalArgumentException;
017    import org.elasticsearch.common.xcontent.ToXContent;
018    import org.elasticsearch.common.xcontent.XContentBuilder;
019    import org.xbib.elasticsearch.action.skywalker.support.IndexableFieldToXContent;
020    
021    import java.io.IOException;
022    import java.util.ArrayList;
023    import java.util.List;
024    
025    import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
026    
027    /**
028     * This class attempts to reconstruct all fields from a document existing in a
029     * Lucene index. This operation may be (and usually) is lossy - e.g. unstored
030     * fields are rebuilt from terms present in the index, and these terms may have
031     * been changed (e.g. lowercased, stemmed), and many other input tokens may have
032     * been skipped altogether by the Analyzer, when fields were originally added to
033     * the index.
034     *
035     */
036    public class DocumentReconstructor {
037    
038        private AtomicReader reader;
039    
040        /**
041         * Prepare a document reconstructor.
042         *
043         * @param indexReader IndexReader to read from.
044         * @throws Exception
045         */
046        public DocumentReconstructor(IndexReader indexReader) {
047            if (indexReader == null) {
048                throw new ElasticsearchIllegalArgumentException("reader cannot be null");
049            }
050            try {
051                if (indexReader instanceof CompositeReader) {
052                    this.reader = SlowCompositeReaderWrapper.wrap(indexReader);
053                } else if (indexReader instanceof AtomicReader) {
054                    this.reader = (AtomicReader) indexReader;
055                } else {
056                    throw new ElasticsearchIllegalArgumentException("unsupported IndexReader class " + indexReader.getClass().getName());
057                }
058            } catch (IOException e) {
059                throw new ElasticsearchIllegalArgumentException(e.getMessage());
060            }
061        }
062    
063        /**
064         * Reconstruct an index shard
065         *
066         * @return reconstructed document
067         * @throws Exception
068         */
069        public XContentBuilder reconstruct(int shardId) throws IOException {
070            XContentBuilder builder = jsonBuilder();
071            builder.startObject()
072                    .field("shardId", shardId)
073                    .field("numDeletions", reader.numDeletedDocs());
074            builder.startArray("docs");
075            FieldInfos fieldInfos = reader.getFieldInfos();
076            Bits live = MultiFields.getLiveDocs(reader);
077            for (int docNum = 0; docNum < reader.maxDoc(); docNum++) {
078                Document doc = reader.document(docNum);
079                if (live != null && live.get(docNum)) {
080                    continue; // not deleted
081                }
082                builder.startObject().startArray("fields");
083                if (fieldInfos != null) {
084                    for (FieldInfo fi : fieldInfos) {
085                        String name = fi.name;
086                        IndexableField[] fs = doc.getFields(name);
087                        if (fs != null && fs.length > 0) {
088                            for (IndexableField f : fs) {
089                                IndexableFieldToXContent x = new IndexableFieldToXContent().field(f);
090                                x.toXContent(builder, ToXContent.EMPTY_PARAMS);
091                            }
092                        }
093                    }
094                }
095                builder.endArray();
096                builder.startArray("terms");
097                if (fieldInfos != null) {
098                    TermsEnum te = null;
099                    DocsAndPositionsEnum dpe = null;
100                    for (FieldInfo fi : fieldInfos) {
101                        Terms terms = MultiFields.getTerms(reader, fi.name);
102                        if (terms == null) { // no terms in this field
103                            continue;
104                        }
105                        te = terms.iterator(te);
106                        while (te.next() != null) {
107                            DocsAndPositionsEnum newDpe = te.docsAndPositions(live, dpe, 0);
108                            if (newDpe == null) { // no position info for this field
109                                break;
110                            }
111                            dpe = newDpe;
112                            int num = dpe.advance(docNum);
113                            if (num != docNum) { // either greater than or NO_MORE_DOCS
114                                continue; // no data for this term in this doc
115                            }
116                            String text = te.term().utf8ToString();
117                            List<Integer> positions = new ArrayList();
118                            List<Integer> starts = new ArrayList();
119                            List<Integer> ends = new ArrayList();
120                            for (int k = 0; k < dpe.freq(); k++) {
121                                int pos = dpe.nextPosition();
122                                positions.add(pos);
123                                starts.add(dpe.startOffset());
124                                ends.add(dpe.endOffset());
125                            }
126                            builder.startObject()
127                                    .field("text", text)
128                                    .field("positions", positions)
129                                    .field("starts", starts)
130                                    .field("ends", ends)
131                                    .field("count", dpe.freq())
132                                    .endObject();
133                        }
134                    }
135                }
136                builder.endArray();
137                builder.endObject();
138            }
139            builder.endArray();
140            builder.endObject();
141            return builder;
142        }
143    
144    }
