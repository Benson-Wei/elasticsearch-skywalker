019package org.xbib.elasticsearch.skywalker.reconstruct;
020
021import org.apache.lucene.document.Document;
022import org.apache.lucene.index.AtomicReader;
023import org.apache.lucene.index.CompositeReader;
024import org.apache.lucene.index.DocsAndPositionsEnum;
025import org.apache.lucene.index.FieldInfo;
026import org.apache.lucene.index.FieldInfos;
027import org.apache.lucene.index.IndexReader;
028import org.apache.lucene.index.IndexableField;
029import org.apache.lucene.index.MultiFields;
030import org.apache.lucene.index.SlowCompositeReaderWrapper;
031import org.apache.lucene.index.Terms;
032import org.apache.lucene.index.TermsEnum;
033import org.apache.lucene.util.Bits;
034import org.elasticsearch.ElasticSearchIllegalArgumentException;
035import org.elasticsearch.common.xcontent.ToXContent;
036import org.elasticsearch.common.xcontent.XContentBuilder;
037import org.xbib.elasticsearch.action.skywalker.support.IndexableFieldToXContent;
038
039import java.io.IOException;
040import java.util.ArrayList;
041import java.util.List;
042
043import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
044
045/**
046 * This class attempts to reconstruct all fields from a document existing in a
047 * Lucene index. This operation may be (and usually) is lossy - e.g. unstored
048 * fields are rebuilt from terms present in the index, and these terms may have
049 * been changed (e.g. lowercased, stemmed), and many other input tokens may have
050 * been skipped altogether by the Analyzer, when fields were originally added to
051 * the index.
052 *
053 *  @author ab
054 *  @author <a href="mailto:joergprante@gmail.com">J&ouml;rg Prante</a>
055 */
056public class DocumentReconstructor {
057
058    private AtomicReader reader;
059
060    /**
061     * Prepare a document reconstructor.
062     *
063     * @param indexReader IndexReader to read from.
064     * @throws Exception
065     */
066    public DocumentReconstructor(IndexReader indexReader) {
067        if (indexReader == null) {
068            throw new ElasticSearchIllegalArgumentException("reader cannot be null");
069        }
070        try {
071            if (indexReader instanceof CompositeReader) {
072                this.reader = new SlowCompositeReaderWrapper((CompositeReader) indexReader);
073            } else if (indexReader instanceof AtomicReader) {
074                this.reader = (AtomicReader) indexReader;
075            } else {
076                throw new ElasticSearchIllegalArgumentException("unsupported IndexReader class " + indexReader.getClass().getName());
077            }
078        } catch (IOException e) {
079            throw new ElasticSearchIllegalArgumentException(e.getMessage());
080        }
081    }
082
083    /**
084     * Reconstruct an index shard
085     *
086     * @return reconstructed document
087     * @throws Exception
088     */
089    public XContentBuilder reconstruct(int shardId) throws IOException {
090        XContentBuilder builder = jsonBuilder();
091        builder.startObject()
092                .field("shardId", shardId)
093                .field("numDeletions", reader.numDeletedDocs());
094        builder.startArray("docs");
095        FieldInfos fieldInfos = reader.getFieldInfos();
096        Bits live = MultiFields.getLiveDocs(reader);
097        for (int docNum = 0; docNum < reader.maxDoc(); docNum++) {
098            Document doc = reader.document(docNum);
099            if (live != null && live.get(docNum)) {
100                continue; // not deleted
101            }
102            builder.startObject().startArray("fields");
103            if (fieldInfos != null) {
104                for (FieldInfo fi : fieldInfos) {
105                    String name = fi.name;
106                    IndexableField[] fs = doc.getFields(name);
107                    if (fs != null && fs.length > 0) {
108                        for (IndexableField f : fs) {
109                            IndexableFieldToXContent x = new IndexableFieldToXContent().field(f);
110                            x.toXContent(builder, ToXContent.EMPTY_PARAMS);
111                        }
112                    }
113                }
114            }
115            builder.endArray();
116            builder.startArray("terms");
117            if (fieldInfos != null) {
118                TermsEnum te = null;
119                DocsAndPositionsEnum dpe = null;
120                for (FieldInfo fi : fieldInfos) {
121                    Terms terms = MultiFields.getTerms(reader, fi.name);
122                    if (terms == null) { // no terms in this field
123                        continue;
124                    }
125                    te = terms.iterator(te);
126                    while (te.next() != null) {
127                        DocsAndPositionsEnum newDpe = te.docsAndPositions(live, dpe, 0);
128                        if (newDpe == null) { // no position info for this field
129                            break;
130                        }
131                        dpe = newDpe;
132                        int num = dpe.advance(docNum);
133                        if (num != docNum) { // either greater than or NO_MORE_DOCS
134                            continue; // no data for this term in this doc
135                        }
136                        String text = te.term().utf8ToString();
137                        List<Integer> positions = new ArrayList();
138                        List<Integer> starts = new ArrayList();
139                        List<Integer> ends = new ArrayList();
140                        for (int k = 0; k < dpe.freq(); k++) {
141                            int pos = dpe.nextPosition();
142                            positions.add(pos);
143                            starts.add(dpe.startOffset());
144                            ends.add(dpe.endOffset());
145                        }
146                        builder.startObject()
147                                .field("text", text)
148                                .field("positions", positions)
149                                .field("starts", starts)
150                                .field("ends", ends)
151                                .field("count", dpe.freq())
152                                .endObject();
153                    }
154                }
155            }
156            builder.endArray();
157            builder.endObject();
158        }
159        builder.endArray();
160        builder.endObject();
161        return builder;
162    }
