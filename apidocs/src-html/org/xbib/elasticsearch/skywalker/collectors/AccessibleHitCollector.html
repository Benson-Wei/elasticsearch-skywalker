002package org.xbib.elasticsearch.skywalker.collectors;
003
004import org.apache.lucene.search.Collector;
005import org.apache.lucene.search.Scorer;
006
007/**
008 *  Accessible Hit collector
009 */
010public abstract class AccessibleHitCollector extends Collector {
011
012    protected Scorer scorer;
013    protected boolean shouldScore;
014    protected int docBase;
015    protected boolean outOfOrder;
016
017    public abstract int getTotalHits();
018
019    public abstract int getDocId(int pos);
020
021    public abstract float getScore(int pos);
022
023    public abstract void reset();
024}
