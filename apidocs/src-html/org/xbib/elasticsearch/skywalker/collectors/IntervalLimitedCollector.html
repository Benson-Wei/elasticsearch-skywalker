001    
002    package org.xbib.elasticsearch.skywalker.collectors;
003    
004    import org.apache.lucene.index.AtomicReaderContext;
005    import org.apache.lucene.search.Scorer;
006    import org.apache.lucene.search.TimeLimitingCollector;
007    import org.apache.lucene.search.TimeLimitingCollector.TimeExceededException;
008    import org.apache.lucene.search.TopDocs;
009    import org.apache.lucene.search.TopScoreDocCollector;
010    
011    import java.io.IOException;
012    
013    /**
014     * Interval limited collector
015     */
016    public class IntervalLimitedCollector extends LimitedHitCollector {
017    
018        private long maxTime;
019        private long lastDoc = 0;
020        private TopScoreDocCollector tdc;
021        private TopDocs topDocs = null;
022        private TimeLimitingCollector thc;
023    
024        public IntervalLimitedCollector(int maxTime, boolean outOfOrder, boolean shouldScore) {
025            this.maxTime = maxTime;
026            this.outOfOrder = outOfOrder;
027            this.shouldScore = shouldScore;
028            tdc = TopScoreDocCollector.create(1000, outOfOrder);
029            thc = new TimeLimitingCollector(tdc, TimeLimitingCollector.getGlobalCounter(), maxTime);
030        }
031    
032        @Override
033        public long limitSize() {
034            return maxTime;
035        }
036    
037        @Override
038        public int limitType() {
039            return TYPE_TIME;
040        }
041    
042        @Override
043        public int getDocId(int pos) {
044            if (topDocs == null) {
045                topDocs = tdc.topDocs();
046            }
047            return topDocs.scoreDocs[pos].doc;
048        }
049    
050        @Override
051        public float getScore(int pos) {
052            if (topDocs == null) {
053                topDocs = tdc.topDocs();
054            }
055            return topDocs.scoreDocs[pos].score;
056        }
057    
058        @Override
059        public int getTotalHits() {
060            return tdc.getTotalHits();
061        }
062    
063        @Override
064        public void collect(int docNum) throws IOException {
065            try {
066                thc.collect(docNum);
067            } catch (TimeExceededException tee) {
068                // re-throw
069                throw new LimitedException(TYPE_TIME, maxTime, tee.getTimeElapsed(), tee.getLastDocCollected());
070            }
071        }
072    
073        @Override
074        public boolean acceptsDocsOutOfOrder() {
075            return outOfOrder;
076        }
077    
078    
079        @Override
080        public void setNextReader(AtomicReaderContext context) throws IOException {
081            this.docBase = context.docBase;
082            thc.setNextReader(context);
083        }
084    
085        @Override
086        public void setScorer(Scorer scorer) throws IOException {
087            this.scorer = scorer;
088            if (shouldScore) {
089                thc.setScorer(scorer);
090            } else {
091                thc.setScorer(NoScoringScorer.INSTANCE);
092            }
093        }
094    
095        @Override
096        public void reset() {
097            lastDoc = 0;
098            tdc = TopScoreDocCollector.create(1000, outOfOrder);
099            thc = new TimeLimitingCollector(tdc, TimeLimitingCollector.getGlobalCounter(), maxTime);
100        }
101    }
