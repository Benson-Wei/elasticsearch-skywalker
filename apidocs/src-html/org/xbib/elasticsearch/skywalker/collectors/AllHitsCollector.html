002    package org.xbib.elasticsearch.skywalker.collectors;
003    
004    import org.apache.lucene.index.AtomicReaderContext;
005    import org.apache.lucene.search.Scorer;
006    
007    import java.io.IOException;
008    import java.util.ArrayList;
009    import java.util.List;
010    
011    public class AllHitsCollector extends AccessibleHitCollector {
012    
013        private List<AllHit> hits = new ArrayList();
014    
015        public AllHitsCollector(boolean outOfOrder, boolean shouldScore) {
016            this.outOfOrder = outOfOrder;
017            this.shouldScore = shouldScore;
018        }
019    
020        @Override
021        public void collect(int doc) {
022            float score = 1.0f;
023            if (shouldScore) {
024                try {
025                    score = scorer.score();
026                } catch (IOException e) {
027                }
028            }
029            hits.add(new AllHit(docBase + doc, score));
030        }
031    
032        @Override
033        public int getTotalHits() {
034            return hits.size();
035        }
036    
037        @Override
038        public int getDocId(int i) {
039            return hits.get(i).docId();
040        }
041    
042        @Override
043        public float getScore(int i) {
044            return hits.get(i).score();
045        }
046    
047        @Override
048        public boolean acceptsDocsOutOfOrder() {
049            return outOfOrder;
050        }
051    
052        @Override
053        public void setNextReader(AtomicReaderContext context) throws IOException {
054            this.docBase = context.docBase;
055        }
056    
057        @Override
058        public void setScorer(Scorer scorer) throws IOException {
059            this.scorer = scorer;
060        }
061    
062        @Override
063        public void reset() {
064            hits.clear();
065        }
066    
067    }
