002    package org.xbib.elasticsearch.skywalker.stats;
003    
004    import org.apache.lucene.util.BytesRef;
005    
006    /**
007     * Term stats
008     */
009    public class TermStats {
010    
011        private String field;
012        private BytesRef termtext;
013        private int docFreq;
014    
015        public TermStats text(BytesRef text) {
016            this.termtext =  text;
017            return this;
018        }
019    
020        public String text() {
021            return termtext.utf8ToString();
022        }
023    
024        public TermStats field(String field) {
025            this.field = field;
026            return this;
027        }
028    
029        public String field() {
030            return field;
031        }
032    
033        public TermStats docFreq(int docFreq) {
034            this.docFreq = docFreq;
035            return this;
036        }
037    
038        public int docFreq() {
039            return docFreq;
040        }
041    
042        public String toString() {
043            return field + ":" + termtext.utf8ToString() + ":" + docFreq;
044        }
045    }
