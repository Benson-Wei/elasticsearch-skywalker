002    package org.xbib.elasticsearch.skywalker.stats;
003    
004    import org.apache.lucene.util.PriorityQueue;
005    
006    /**
007     *  Term stats queue
008     */
009    public class TermStatsQueue extends PriorityQueue<TermStats> {
010    
011        public TermStatsQueue(int size) {
012            super(size);
013        }
014    
015        @Override
016        protected boolean lessThan(TermStats termInfoA, TermStats termInfoB) {
017            return termInfoA.docFreq() < termInfoB.docFreq();
018        }
019    }
