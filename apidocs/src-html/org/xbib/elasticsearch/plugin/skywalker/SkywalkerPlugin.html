002    package org.xbib.elasticsearch.plugin.skywalker;
003    
004    import org.elasticsearch.action.ActionModule;
005    import org.elasticsearch.plugins.AbstractPlugin;
006    import org.elasticsearch.rest.RestModule;
007    import org.xbib.elasticsearch.action.admin.cluster.state.ConsistencyCheckAction;
008    import org.xbib.elasticsearch.action.admin.cluster.state.TransportConsistencyCheckAction;
009    import org.xbib.elasticsearch.action.admin.indices.reconstruct.ReconstructIndexAction;
010    import org.xbib.elasticsearch.action.admin.indices.reconstruct.TransportReconstructAction;
011    import org.xbib.elasticsearch.action.skywalker.SkywalkerAction;
012    import org.xbib.elasticsearch.action.skywalker.TransportSkywalkerAction;
013    import org.xbib.elasticsearch.rest.action.skywalker.RestConsistencyCheckAction;
014    import org.xbib.elasticsearch.rest.action.skywalker.RestReconstructIndexAction;
015    import org.xbib.elasticsearch.rest.action.skywalker.RestSkywalkerAction;
016    
017    /**
018     *  Skywalker plugin
019     */
020    public class SkywalkerPlugin extends AbstractPlugin {
021    
022        @Override
023        public String name() {
024            return "skywalker";
025        }
026    
027        @Override
028        public String description() {
029            return "Skywalker - Luke for Elasticsearch";
030        }
031    
032        public void onModule(RestModule module) {
033            module.addRestAction(RestSkywalkerAction.class);
034            module.addRestAction(RestConsistencyCheckAction.class);
035            module.addRestAction(RestReconstructIndexAction.class);
036        }
037    
038        public void onModule(ActionModule module) {
039            module.registerAction(SkywalkerAction.INSTANCE, TransportSkywalkerAction.class);
040            module.registerAction(ConsistencyCheckAction.INSTANCE, TransportConsistencyCheckAction.class);
041            module.registerAction(ReconstructIndexAction.INSTANCE, TransportReconstructAction.class);
042        }
043    
044    }
