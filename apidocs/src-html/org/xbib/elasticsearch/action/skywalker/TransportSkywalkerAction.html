045/**
046 * Transport action for Skywalker plugin.
047 */
048public class TransportSkywalkerAction
049        extends TransportBroadcastOperationAction<SkywalkerRequest, SkywalkerResponse, ShardSkywalkerRequest, ShardSkywalkerResponse> {
050
051    private final IndicesService indicesService;
052
053    private final NodeEnvironment nodeEnv;
054
055    private final Object mutex = new Object();
056
057    @Inject
058    public TransportSkywalkerAction(Settings settings, ThreadPool threadPool,
059                                    ClusterService clusterService,
060                                    TransportService transportService,
061                                    IndicesService indicesService,
062                                    NodeEnvironment nodeEnv) {
063        super(settings, threadPool, clusterService, transportService);
064        this.indicesService = indicesService;
065        this.nodeEnv = nodeEnv;
066    }
067
068    @Override
069    protected String executor() {
070        return ThreadPool.Names.MERGE;
071    }
072

074    protected String transportAction() {
075        return SkywalkerAction.NAME;

080        return new SkywalkerRequest();

083    @Override
084    protected SkywalkerResponse newResponse(SkywalkerRequest request, AtomicReferenceArray shardsResponses, ClusterState clusterState) {
085        int successfulShards = 0;
086        int failedShards = 0;
087        List<ShardOperationFailedException> shardFailures = null;
088        Map<String, Map<String, Map<String, Object>>> response = new HashMap();
089        for (int i = 0; i < shardsResponses.length(); i++) {
090            Object shardResponse = shardsResponses.get(i);
091            if (shardResponse == null) {
092                // a non active shard, ignore...
093            } else if (shardResponse instanceof BroadcastShardOperationFailedException) {
094                failedShards++;
095                if (shardFailures == null) {
096                    shardFailures = newArrayList();
097                }
098                shardFailures.add(new DefaultShardOperationFailedException((BroadcastShardOperationFailedException) shardResponse));
099            } else {
100                successfulShards++;
101                if (shardResponse instanceof ShardSkywalkerResponse) {
102                    ShardSkywalkerResponse shardResp = (ShardSkywalkerResponse) shardResponse;
103                    String index = shardResp.getIndex();
104                    int shardId = shardResp.getShardId();
105                    // one map per index
106                    Map indexresponse = response.get(index);
107                    if (indexresponse == null) {
108                        indexresponse = new HashMap();
109                    }
110                    // merge index-wide fieldInfo into single field
111                    //indexresponse.put("fieldInfos", shardResp.getResponse().get("fieldInfos"));
112                    //shardResp.getResponse().remove("fieldInfos");
113                    indexresponse.put(Integer.toString(shardId), shardResp.getResponse());
114                    response.put(index, indexresponse);
115                }
116            }
117        }
118        return new SkywalkerResponse(shardsResponses.length(), successfulShards, failedShards, shardFailures).setResponse(response);
119    }
120

124    }

131    @Override

136    @Override
137    protected ShardSkywalkerResponse shardOperation(ShardSkywalkerRequest request) throws ElasticSearchException {
138        synchronized (mutex) {
139            IndexService indexService = indicesService.indexServiceSafe(request.index());
140            InternalIndexShard indexShard = (InternalIndexShard) indexService.shardSafe(request.shardId());
141            MapperService mapperService = indexService.mapperService();
142            Engine.Searcher searcher = indexShard.acquireSearcher("skywalker_action");
143            try {
144                IndexReader reader = searcher.reader();
145
146                Skywalker skywalker = new Skywalker(reader);
147
148                Map<String, Object> response = new HashMap();
149
150                Directory directory = indexShard.store().directory();
151                List indexFiles = new ArrayList();
152                for (String f : skywalker.getIndexFiles(directory)) {
153                    Map indexFile = new HashMap();
154                    indexFile.put("name", f);
155                    indexFile.put("function", skywalker.getFileFunction(f));
156                    indexFiles.add(indexFile);
157                }
158                response.put("indexFiles", indexFiles);
159
160                skywalker.getStoreMetadata(response, indexShard.store().list());
161
162                response.put("indexVersion", skywalker.getVersion());
163                response.put("directoryImpl", skywalker.getDirImpl());
164                response.put("numDocs", reader.numDocs());
165                response.put("maxDoc", reader.maxDoc());
166                response.put("hasDeletions", reader.hasDeletions());
167                response.put("numDeletedDocs", reader.numDeletedDocs());
168
169                Set<FieldTermCount> ftc = skywalker.getFieldTermCounts();
170                response.put("numTerms", skywalker.getNumTerms());
171
172                Map indexFormatInfo = new HashMap();
173                FormatDetails details = skywalker.getFormatDetails();
174                indexFormatInfo.put("version", details.getVersion());
175                indexFormatInfo.put("genericName", details.getGenericName());
176                indexFormatInfo.put("capabilities", details.getCapabilities());
177                response.put("indexFormat", indexFormatInfo);
178
179                List commits = new ArrayList();
180                Iterator<Segment> it = indexShard.engine().segments().iterator();
181                while (it.hasNext()) {
182                    Segment segment = it.next();
183                    Map m = new HashMap();
184                    m.put("segment", segment.getName());
185                    m.put("count", segment.getNumDocs());
186                    m.put("deleted", segment.getDeletedDocs());
187                    m.put("generation", segment.getGeneration());
188                    m.put("sizeInBytes", segment.getSizeInBytes());
189                    m.put("version", segment.getVersion());
190                    m.put("committed", segment.committed);
191                    m.put("compound", segment.compound);
192                    m.put("size", segment.getSize().toString());
193                    commits.add(m);
194                }
195                response.put("commits", commits);
196
197                List fieldInfos = new ArrayList();
198                for (FieldInfo fi : MultiFields.getMergedFieldInfos(reader)) {
199                    fieldInfos.add(skywalker.getFieldInfo(mapperService, fi));
200                }
201                response.put("fieldInfos", fieldInfos);
202
203                List termList = new ArrayList();
204                for (TermStats ts : skywalker.getTopTerms(50)) {
205                    Map m = new HashMap();
206                    m.put("field", ts.field());
207                    m.put("text", ts.text());
208                    m.put("docFreq", ts.docFreq());
209                    termList.add(m);
210                }
211                response.put("topterms", termList);
212                return new ShardSkywalkerResponse(request.index(), request.shardId()).setResponse(response);
213            } catch (Exception ex) {
214                throw new ElasticSearchException(ex.getMessage(), ex);
215            } finally {
216                searcher.release();
217            }
218        }
219    }
220
221    @Override
222    protected GroupShardsIterator shards(ClusterState clusterState, SkywalkerRequest request, String[] concreteIndices) {
223        return clusterState.routingTable().activePrimaryShardsGrouped(concreteIndices, true);
224    }

229    }

233        return state.blocks().globalBlockedException(ClusterBlockLevel.READ);
