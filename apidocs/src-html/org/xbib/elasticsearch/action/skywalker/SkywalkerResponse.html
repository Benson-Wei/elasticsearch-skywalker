002    package org.xbib.elasticsearch.action.skywalker;
003    
004    import org.elasticsearch.action.ShardOperationFailedException;
005    import org.elasticsearch.action.support.broadcast.BroadcastOperationResponse;
006    import org.elasticsearch.common.io.stream.StreamInput;
007    import org.elasticsearch.common.io.stream.StreamOutput;
008    
009    import java.io.IOException;
010    import java.util.HashMap;
011    import java.util.List;
012    import java.util.Map;
013    
014    /**
015     * A response for a skywalker action.
016     */
017    public class SkywalkerResponse extends BroadcastOperationResponse {
018    
019        private Map<String, Map<String, Map<String, Object>>> response;
020    
021        SkywalkerResponse() {
022        }
023    
024        SkywalkerResponse(int totalShards, int successfulShards, int failedShards, List<ShardOperationFailedException> shardFailures) {
025            super(totalShards, successfulShards, failedShards, shardFailures);
026        }
027    
028        public SkywalkerResponse setResponse(Map<String, Map<String, Map<String, Object>>> response) {
029            this.response = response;
030            return this;
031        }
032    
033        public Map<String, Map<String, Map<String, Object>>> getResponse() {
034            return response;
035        }
036    
037        @Override
038        public void readFrom(StreamInput in) throws IOException {
039            super.readFrom(in);
040            int indexCount = in.readInt();
041            this.response = new HashMap();
042            for (int i = 0; i < indexCount; i++) {
043                String index = in.readString();
044                Map<String, Map<String, Object>> shards = new HashMap();
045                int shardCount = in.readInt();
046                for (int j = 0; j < shardCount; j++) {
047                    String shard = in.readString();
048                    Map<String, Object> fields = in.readMap();
049                    shards.put(shard, fields);
050                }
051                response.put(index, shards);
052            }
053        }
054    
055        @Override
056        public void writeTo(StreamOutput out) throws IOException {
057            super.writeTo(out);
058            int indexCount = response.keySet().size();
059            out.writeInt(indexCount);
060            for (String index : response.keySet()) {
061                out.writeString(index);
062                int shardCount = response.get(index).keySet().size();
063                out.writeInt(shardCount);
064                for (String shard : response.get(index).keySet()) {
065                    out.writeString(shard);
066                    out.writeMap(response.get(index).get(shard));
067                }
068            }
069        }
070    }
