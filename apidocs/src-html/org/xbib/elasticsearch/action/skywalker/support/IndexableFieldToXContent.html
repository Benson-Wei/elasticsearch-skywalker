002package org.xbib.elasticsearch.action.skywalker.support;
003
004import org.apache.lucene.document.DoubleField;
005import org.apache.lucene.document.FloatField;
006import org.apache.lucene.document.IntField;
007import org.apache.lucene.document.LongField;
008import org.apache.lucene.index.IndexableField;
009import org.elasticsearch.common.xcontent.ToXContent;
010import org.elasticsearch.common.xcontent.XContentBuilder;
011
012import java.io.IOException;
013
014/**
015 *  Indexable field to XContent
016 */
017public class IndexableFieldToXContent implements ToXContent {
018
019    private IndexableField field;
020
021    public IndexableFieldToXContent field(IndexableField field) {
022        this.field = field;
023        return this;
024    }
025
026    @Override
027    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
028        builder.startObject();
029        builder.field("name", field.name());
030        if (field.binaryValue() != null) {
031            builder.field("type", "binary")
032                    .field("value", field.binaryValue());
033        } else if (field.stringValue() != null) {
034            builder.field("type", "string")
035                    .field("value", field.stringValue());
036        } else if (field.numericValue() != null) {
037            if (field instanceof IntField) {
038                builder.field("type", "integer")
039                        .field("value", field.numericValue().intValue());
040            } else if (field instanceof LongField) {
041                builder.field("type", "long")
042                        .field("value", field.numericValue().longValue());
043            } else if (field instanceof FloatField) {
044                builder.field("type", "float")
045                        .field("value", field.numericValue().floatValue());
046            } else if (field instanceof DoubleField) {
047                builder.field("type", "double")
048                        .field("value", field.numericValue().doubleValue());
049            }
050        }
051        builder.field("stored", field.fieldType().stored());
052        builder.field("indexed", field.fieldType().indexed());
053        builder.field("omitNorms", field.fieldType().omitNorms());
054        builder.field("storeTermVectors", field.fieldType().storeTermVectors());
055        builder.field("storeTermVectorOffsets", field.fieldType().storeTermVectorOffsets());
056        builder.field("storeTermVectorPayloads", field.fieldType().storeTermVectorPayloads());
057        builder.field("storeTermVectorPositions", field.fieldType().storeTermVectorPositions());
058        builder.field("boost", field.boost());
059        builder.endObject();
060        return builder;
061    }
062}
