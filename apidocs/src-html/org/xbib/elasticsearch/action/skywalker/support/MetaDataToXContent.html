002package org.xbib.elasticsearch.action.skywalker.support;
003
004import org.elasticsearch.cluster.metadata.AliasMetaData;
005import org.elasticsearch.cluster.metadata.IndexMetaData;
006import org.elasticsearch.cluster.metadata.IndexTemplateMetaData;
007import org.elasticsearch.cluster.metadata.MappingMetaData;
008import org.elasticsearch.cluster.metadata.MetaData;
009import org.elasticsearch.common.xcontent.ToXContent;
010import org.elasticsearch.common.xcontent.XContentBuilder;
011
012import java.io.IOException;
013import java.util.Map;
014
015/**
016 * Metadata to XContent
017 */
018public class MetaDataToXContent implements ToXContent {
019
020    MetaData metadata;
021
022    public MetaDataToXContent metadata(MetaData metadata) {
023        this.metadata = metadata;
024        return this;
025    }
026
027    @Override
028    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
029        builder.startObject();
030        builder.field("version", metadata.version());
031        builder.field("numberOfShards", metadata.numberOfShards());
032        builder.startArray("concreteIndices");
033        for (String index : metadata.concreteAllIndicesAsSet()) {
034            builder.value(index);
035        }
036        builder.endArray();
037        Map<String, Map<String, AliasMetaData>> aliases = metadata.getAliases();
038        builder.startArray("aliases");
039        for (String alias : aliases.keySet()) {
040            builder.startObject(alias);
041            builder.startArray("aliasMetadata");
042            for (String s : aliases.get(alias).keySet()) {
043                builder.startObject();
044                AliasMetaData amd = aliases.get(alias).get(s);
045                builder.field("alias", amd.getAlias());
046                builder.field("filter", amd.getFilter().string());
047                builder.field("indexRouting", amd.getIndexRouting());
048                builder.field("searchRouting", amd.getSearchRouting());
049                builder.endObject();
050            }
051            builder.endArray();
052            builder.endObject();
053        }
054        builder.endArray();
055        builder.startArray("indexes");
056        Map<String,IndexMetaData> indices = metadata.getIndices();
057        for (String s : indices.keySet()) {
058            IndexMetaData imd = indices.get(s);
059            builder.startObject();
060            builder.field("index", imd.getIndex());
061            builder.field("state", imd.getState().name());
062            builder.field("numberOfReplicas", imd.getNumberOfReplicas());
063            builder.field("numberOfShards", imd.getNumberOfShards());
064            builder.field("totalNumberOfShards", imd.getTotalNumberOfShards());
065            builder.field("version", imd.getVersion());
066            builder.field("settings", imd.getSettings().getAsMap());
067            Map<String, MappingMetaData> m = imd.getMappings();
068            // skip mappings here
069            builder.endObject();
070        }
071        builder.endArray();
072        builder.startArray("templates");
073        Map<String,IndexTemplateMetaData> templates = metadata.getTemplates();
074        for (String s : templates.keySet()) {
075            IndexTemplateMetaData itmd = templates.get(s);
076            itmd.getName();
077            itmd.getOrder();
078            itmd.getTemplate();
079            itmd.getSettings();
080            itmd.getMappings();
081        }
082        builder.endArray();
083        builder.endObject();
084        return builder;
085    }
086}
