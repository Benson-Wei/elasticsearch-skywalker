002    package org.xbib.elasticsearch.action.skywalker.support;
003    
004    import org.elasticsearch.cluster.metadata.AliasMetaData;
005    import org.elasticsearch.cluster.metadata.IndexMetaData;
006    import org.elasticsearch.cluster.metadata.IndexTemplateMetaData;
007    import org.elasticsearch.cluster.metadata.MappingMetaData;
008    import org.elasticsearch.cluster.metadata.MetaData;
009    import org.elasticsearch.common.collect.ImmutableOpenMap;
010    import org.elasticsearch.common.hppc.cursors.ObjectCursor;
011    import org.elasticsearch.common.xcontent.ToXContent;
012    import org.elasticsearch.common.xcontent.XContentBuilder;
013    
014    import java.io.IOException;
015    import java.util.Map;
016    
017    /**
018     * Metadata to XContent
019     */
020    public class MetaDataToXContent implements ToXContent {
021    
022        MetaData metadata;
023    
024        public MetaDataToXContent metadata(MetaData metadata) {
025            this.metadata = metadata;
026            return this;
027        }
028    
029        @Override
030        public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
031            builder.startObject();
032            builder.field("version", metadata.version());
033            builder.field("numberOfShards", metadata.numberOfShards());
034            builder.startArray("concreteIndices");
035            for (String index : metadata.concreteAllIndices()) {
036                builder.value(index);
037            }
038            builder.endArray();
039            ImmutableOpenMap<String, ImmutableOpenMap<String, AliasMetaData>> aliases = metadata.getAliases();
040            builder.startArray("aliases");
041            for (ObjectCursor<String> alias : aliases.keys()) {
042                builder.startObject(alias.value);
043                builder.startArray("aliasMetadata");
044                for (ObjectCursor<String> s : aliases.get(alias.value).keys()) {
045                    builder.startObject();
046                    AliasMetaData amd = aliases.get(alias.value).get(s.value);
047                    builder.field("alias", amd.getAlias());
048                    builder.field("filter", amd.getFilter().string());
049                    builder.field("indexRouting", amd.getIndexRouting());
050                    builder.field("searchRouting", amd.getSearchRouting());
051                    builder.endObject();
052                }
053                builder.endArray();
054                builder.endObject();
055            }
056            builder.endArray();
057            builder.startArray("indexes");
058            ImmutableOpenMap<String,IndexMetaData> indices = metadata.getIndices();
059            for (ObjectCursor<String> s : indices.keys()) {
060                IndexMetaData imd = indices.get(s.value);
061                builder.startObject();
062                builder.field("index", imd.getIndex());
063                builder.field("state", imd.getState().name());
064                builder.field("numberOfReplicas", imd.getNumberOfReplicas());
065                builder.field("numberOfShards", imd.getNumberOfShards());
066                builder.field("totalNumberOfShards", imd.getTotalNumberOfShards());
067                builder.field("version", imd.getVersion());
068                builder.field("settings", imd.getSettings().getAsMap());
069                ImmutableOpenMap<String, MappingMetaData> m = imd.getMappings();
070                // skip mappings here
071                builder.endObject();
072            }
073            builder.endArray();
074            builder.startArray("templates");
075            ImmutableOpenMap<String,IndexTemplateMetaData> templates = metadata.getTemplates();
076            for (ObjectCursor<String> s : templates.keys()) {
077                IndexTemplateMetaData itmd = templates.get(s.value);
078                itmd.getName();
079                itmd.getOrder();
080                itmd.getTemplate();
081                itmd.getSettings();
082                itmd.getMappings();
083            }
084            builder.endArray();
085            builder.endObject();
086            return builder;
087        }
088    }
