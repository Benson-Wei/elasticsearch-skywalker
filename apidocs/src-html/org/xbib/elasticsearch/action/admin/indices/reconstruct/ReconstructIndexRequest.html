002package org.xbib.elasticsearch.action.admin.indices.reconstruct;
003
004import org.elasticsearch.action.support.broadcast.BroadcastOperationRequest;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007
008import java.io.IOException;
009
010public class ReconstructIndexRequest extends BroadcastOperationRequest<ReconstructIndexRequest> {
011
012    private String index;
013
014    ReconstructIndexRequest() {
015    }
016
017    public ReconstructIndexRequest(String index) {
018        this.index = index;
019    }
020
021    public String index() {
022        return index;
023    }
024
025    @Override
026    public void readFrom(StreamInput in) throws IOException {
027        super.readFrom(in);
028        this.index = in.readString();
029    }
030
031    @Override
032    public void writeTo(StreamOutput out) throws IOException {
033        super.writeTo(out);
034        out.writeString(index);
035    }
036}
