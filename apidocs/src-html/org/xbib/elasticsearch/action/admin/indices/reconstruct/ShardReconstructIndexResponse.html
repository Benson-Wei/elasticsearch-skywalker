002    package org.xbib.elasticsearch.action.admin.indices.reconstruct;
003    
004    import org.elasticsearch.action.support.broadcast.BroadcastShardOperationResponse;
005    import org.elasticsearch.common.io.stream.StreamInput;
006    import org.elasticsearch.common.io.stream.StreamOutput;
007    import org.elasticsearch.common.xcontent.XContentBuilder;
008    import org.elasticsearch.common.xcontent.XContentHelper;
009    import org.elasticsearch.common.xcontent.XContentParser;
010    
011    import java.io.IOException;
012    
013    import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
014    
015    public class ShardReconstructIndexResponse extends BroadcastShardOperationResponse {
016    
017        private boolean exists;
018        private XContentBuilder builder;
019    
020        ShardReconstructIndexResponse() {
021        }
022    
023        ShardReconstructIndexResponse(boolean exists) {
024            this.exists = exists;
025        }
026    
027        ShardReconstructIndexResponse(boolean exists, XContentBuilder builder) {
028            this.exists = exists;
029            this.builder = builder;
030        }
031    
032        public XContentBuilder getReconstructedIndex() {
033            return builder;
034        }
035    
036        @Override
037        public void readFrom(StreamInput in) throws IOException {
038            super.readFrom(in);
039            exists = in.readBoolean();
040            if (in.readBoolean()) {
041                builder = jsonBuilder();
042                XContentParser p = XContentHelper.createParser(in.readBytesReference());
043                builder.copyCurrentStructure(p);
044            }
045        }
046    
047        @Override
048        public void writeTo(StreamOutput out) throws IOException {
049            super.writeTo(out);
050            out.writeBoolean(exists);
051            if (builder == null) {
052                out.writeBoolean(false);
053            } else {
054                out.writeBoolean(true);
055                out.writeBytesReference(builder.bytes());
056            }
057        }
058    }
