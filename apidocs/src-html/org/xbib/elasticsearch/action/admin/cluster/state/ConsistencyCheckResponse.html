001    
002    package org.xbib.elasticsearch.action.admin.cluster.state;
003    
004    import org.elasticsearch.action.ActionResponse;
005    import org.elasticsearch.cluster.ClusterName;
006    import org.elasticsearch.cluster.ClusterState;
007    import org.elasticsearch.common.io.stream.StreamInput;
008    import org.elasticsearch.common.io.stream.StreamOutput;
009    
010    import java.io.File;
011    import java.io.IOException;
012    import java.util.ArrayList;
013    import java.util.List;
014    
015    /**
016     *  Consistency check response
017     */
018    public class ConsistencyCheckResponse extends ActionResponse {
019    
020        private ClusterName clusterName;
021    
022        private ClusterState clusterState;
023    
024        private List<File> files;
025    
026        public ConsistencyCheckResponse() {
027        }
028    
029        ConsistencyCheckResponse(ClusterName clusterName, ClusterState clusterState, List<File> files) {
030            this.clusterName = clusterName;
031            this.clusterState = clusterState;
032            this.files = files;
033        }
034    
035        public ClusterState getState() {
036            return this.clusterState;
037        }
038    
039        public ClusterName getClusterName() {
040            return this.clusterName;
041        }
042    
043        public List<File> getFiles() {
044            return this.files;
045        }
046    
047        @Override
048        public void readFrom(StreamInput in) throws IOException {
049            super.readFrom(in);
050            clusterName = ClusterName.readClusterName(in);
051            clusterState = ClusterState.Builder.readFrom(in, null);
052            int n = in.read();
053            files = new ArrayList();
054            for (int i = 0; i < n; i++) {
055                files.set(i, new File(in.readString()));
056            }
057        }
058    
059        @Override
060        public void writeTo(StreamOutput out) throws IOException {
061            super.writeTo(out);
062            clusterName.writeTo(out);
063            ClusterState.Builder.writeTo(clusterState, out);
064            out.write(files.size());
065            for (File file : files) {
066                out.writeString(file.getAbsolutePath());
067            }
068        }
069    }
