002package org.xbib.elasticsearch.action.admin.cluster.state;
003
004import org.elasticsearch.ElasticSearchException;
005import org.elasticsearch.action.ActionListener;
006import org.elasticsearch.action.support.master.TransportMasterNodeOperationAction;
007import org.elasticsearch.cluster.ClusterName;
008import org.elasticsearch.cluster.ClusterService;
009import org.elasticsearch.cluster.ClusterState;
010import org.elasticsearch.common.inject.Inject;
011import org.elasticsearch.common.settings.Settings;
012import org.elasticsearch.env.NodeEnvironment;
013import org.elasticsearch.threadpool.ThreadPool;
014import org.elasticsearch.transport.TransportService;
015import org.xbib.elasticsearch.skywalker.Skywalker;
016
017import java.io.File;
018import java.util.ArrayList;
019import java.util.List;
020
021import static org.elasticsearch.cluster.ClusterState.builder;
022
023/**
024 *  Transport consistency check action
025 *
026 */
027public class TransportConsistencyCheckAction extends TransportMasterNodeOperationAction<ConsistencyCheckRequest, ConsistencyCheckResponse> {
028
029    private final ClusterName clusterName;
030
031    private final NodeEnvironment nodeEnv;
032
033    @Inject
034    public TransportConsistencyCheckAction(Settings settings, TransportService transportService, ClusterService clusterService, ThreadPool threadPool,
035                                           ClusterName clusterName, NodeEnvironment nodeEnvironment) {
036        super(settings, transportService, clusterService, threadPool);
037        this.clusterName = clusterName;
038        this.nodeEnv = nodeEnvironment;
039    }
040
041    @Override
042    protected String executor() {
043        return ThreadPool.Names.GENERIC;
044    }
045
046    @Override
047    protected String transportAction() {
048        return ConsistencyCheckAction.NAME;
049    }
050
051    @Override
052    protected ConsistencyCheckRequest newRequest() {
053        return new ConsistencyCheckRequest();
054    }
055
056    @Override
057    protected ConsistencyCheckResponse newResponse() {
058        return new ConsistencyCheckResponse();
059    }
060
061    @Override
062    protected void masterOperation(ConsistencyCheckRequest request, ClusterState state, ActionListener<ConsistencyCheckResponse> listener) throws ElasticSearchException {
063        ClusterState.Builder builder = builder();
064        List<File> files = new ArrayList();
065        builder.metaData(Skywalker.loadState(files, nodeEnv));
066        listener.onResponse(new ConsistencyCheckResponse(clusterName, builder.build(), files));
067    }
068
069    @Override
070    protected boolean localExecute(ConsistencyCheckRequest request) {
071        return true;
072    }
073
074}
