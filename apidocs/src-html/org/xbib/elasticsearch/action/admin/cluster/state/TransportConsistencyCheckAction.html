049/**
050 *  Transport consistency check action
051 *
052 *  @author <a href="mailto:joergprante@gmail.com">J&ouml;rg Prante</a>
053 */
054public class TransportConsistencyCheckAction extends TransportMasterNodeOperationAction<ConsistencyCheckRequest, ConsistencyCheckResponse> {
055
056    private final ClusterName clusterName;
057
058    private final NodeEnvironment nodeEnv;
059
060    @Inject
061    public TransportConsistencyCheckAction(Settings settings, TransportService transportService, ClusterService clusterService, ThreadPool threadPool,
062                                           ClusterName clusterName, NodeEnvironment nodeEnvironment) {
063        super(settings, transportService, clusterService, threadPool);
064        this.clusterName = clusterName;
065        this.nodeEnv = nodeEnvironment;
066    }
067
068    @Override
069    protected String executor() {
070        return ThreadPool.Names.GENERIC;
071    }
072
073    @Override
074    protected String transportAction() {
075        return ConsistencyCheckAction.NAME;
076    }
077
078    @Override
079    protected ConsistencyCheckRequest newRequest() {
080        return new ConsistencyCheckRequest();
081    }
082
083    @Override
084    protected ConsistencyCheckResponse newResponse() {
085        return new ConsistencyCheckResponse();
086    }
087
088    @Override
089    protected boolean localExecute(ConsistencyCheckRequest request) {
090        return true;
091    }
092
093    @Override
094    protected ConsistencyCheckResponse masterOperation(ConsistencyCheckRequest request, ClusterState state) throws ElasticSearchException {
095        ClusterState.Builder builder = newClusterStateBuilder();
096        List<File> files = new ArrayList();
097        builder.metaData(Skywalker.loadState(files, nodeEnv));
098        return new ConsistencyCheckResponse(clusterName, builder.build(), files);
099    }
100
101}
