002package org.xbib.elasticsearch.rest.action.skywalker;
003
004import org.elasticsearch.action.ActionListener;
005import org.elasticsearch.client.Client;
006import org.elasticsearch.common.inject.Inject;
007import org.elasticsearch.common.joda.time.Instant;
008import org.elasticsearch.common.settings.Settings;
009import org.elasticsearch.common.unit.SizeUnit;
010import org.elasticsearch.common.unit.SizeValue;
011import org.elasticsearch.common.xcontent.ToXContent;
012import org.elasticsearch.common.xcontent.XContentBuilder;
013import org.elasticsearch.rest.BaseRestHandler;
014import org.elasticsearch.rest.RestChannel;
015import org.elasticsearch.rest.RestController;
016import org.elasticsearch.rest.RestRequest;
017import org.elasticsearch.rest.XContentRestResponse;
018import org.elasticsearch.rest.XContentThrowableRestResponse;
019import org.elasticsearch.rest.action.support.RestXContentBuilder;
020import org.xbib.elasticsearch.action.admin.cluster.state.ConsistencyCheckAction;
021import org.xbib.elasticsearch.action.admin.cluster.state.ConsistencyCheckRequest;
022import org.xbib.elasticsearch.action.admin.cluster.state.ConsistencyCheckResponse;
023
024import java.io.File;
025import java.io.IOException;
026
027import static org.elasticsearch.rest.RestRequest.Method.GET;
028import static org.elasticsearch.rest.RestStatus.OK;
029
030/**
031 * REST consistency check action
032 */
033public class RestConsistencyCheckAction extends BaseRestHandler {
034
035    @Inject
036    public RestConsistencyCheckAction(Settings settings, Client client, RestController controller) {
037        super(settings, client);
038        controller.registerHandler(GET, "/_skywalker/consistencycheck", this);
039    }
040
041    @Override
042    public void handleRequest(final RestRequest request, final RestChannel channel) {
043        ConsistencyCheckRequest r = new ConsistencyCheckRequest();
044        client.admin().cluster().execute(ConsistencyCheckAction.INSTANCE, r, new ActionListener<ConsistencyCheckResponse>() {
045
046            @Override
047            public void onResponse(ConsistencyCheckResponse response) {
048                try {
049                    XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
050                    builder.startObject();
051                    builder.field("ok", true);
052                    builder.startObject("state");
053                    response.getState().toXContent(builder, ToXContent.EMPTY_PARAMS);
054                    builder.startArray("files");
055                    for (File file : response.getFiles()) {
056                        Instant instant = new Instant(file.lastModified());
057                        builder.startObject()
058                            .field("path", file.getAbsolutePath())
059                            .field("lastmodified", instant.toDateTime().toString())
060                            .field("size", new SizeValue(file.length(), SizeUnit.SINGLE).toString())
061                            .field("totalspace", new SizeValue(file.getTotalSpace(), SizeUnit.SINGLE).toString())
062                            .field("usablespace", new SizeValue(file.getUsableSpace(), SizeUnit.SINGLE).toString())
063                            .field("freespace", new SizeValue(file.getFreeSpace(), SizeUnit.SINGLE).toString())
064                        .endObject();
065                    }
066                    builder.endArray();
067                    builder.endObject();
068                    channel.sendResponse(new XContentRestResponse(request, OK, builder));
069                } catch (Exception e) {
070                    onFailure(e);
071                }
072            }
073
074            @Override
075            public void onFailure(Throwable e) {
076                try {
077                    logger.error(e.getMessage(), e);
078                    channel.sendResponse(new XContentThrowableRestResponse(request, e));
079                } catch (IOException e1) {
080                    logger.error("Failed to send failure response", e1);
081                }
082            }
083        });
084    }
085}
