002    package org.xbib.elasticsearch.rest.action.skywalker;
003    
004    import org.elasticsearch.action.ActionListener;
005    import org.elasticsearch.client.Client;
006    import org.elasticsearch.common.inject.Inject;
007    import org.elasticsearch.common.settings.Settings;
008    import org.elasticsearch.common.xcontent.XContentBuilder;
009    import org.elasticsearch.common.xcontent.XContentHelper;
010    import org.elasticsearch.common.xcontent.XContentParser;
011    import org.elasticsearch.rest.BaseRestHandler;
012    import org.elasticsearch.rest.RestChannel;
013    import org.elasticsearch.rest.RestController;
014    import org.elasticsearch.rest.RestRequest;
015    import org.elasticsearch.rest.XContentRestResponse;
016    import org.elasticsearch.rest.XContentThrowableRestResponse;
017    import org.elasticsearch.rest.action.support.RestXContentBuilder;
018    import org.xbib.elasticsearch.action.admin.indices.reconstruct.ReconstructIndexAction;
019    import org.xbib.elasticsearch.action.admin.indices.reconstruct.ReconstructIndexRequest;
020    import org.xbib.elasticsearch.action.admin.indices.reconstruct.ReconstructIndexResponse;
021    import org.xbib.elasticsearch.action.admin.indices.reconstruct.ShardReconstructIndexResponse;
022    
023    import java.io.IOException;
024    
025    import static org.elasticsearch.rest.RestRequest.Method.GET;
026    import static org.elasticsearch.rest.RestStatus.OK;
027    
028    /**
029     *  REST action for reconstructing an index
030     */
031    public class RestReconstructIndexAction extends BaseRestHandler {
032    
033        @Inject
034        public RestReconstructIndexAction(Settings settings, Client client, RestController controller) {
035            super(settings, client);
036            controller.registerHandler(GET, "/{index}/_skywalker/reconstruct", this);
037        }
038    
039        @Override
040        public void handleRequest(final RestRequest request, final RestChannel channel) {
041            client.admin().indices().execute(ReconstructIndexAction.INSTANCE,
042                    new ReconstructIndexRequest(request.param("index")),
043                    new ActionListener<ReconstructIndexResponse>() {
044                        @Override
045                        public void onResponse(ReconstructIndexResponse response) {
046                            try {
047                                XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
048                                builder.startObject()
049                                        .field("ok", true)
050                                        .field("index", request.param("index"))
051                                        .startArray("shards");
052                                for (ShardReconstructIndexResponse r : response.shards()) {
053                                    XContentParser p = XContentHelper.createParser(r.getReconstructedIndex().bytes());
054                                    builder.copyCurrentStructure(p);
055                                }
056                                builder.endArray();
057                                builder.endObject();
058                                channel.sendResponse(new XContentRestResponse(request, OK, builder));
059                            } catch (Exception e) {
060                                onFailure(e);
061                            }
062                        }
063    
064                        @Override
065                        public void onFailure(Throwable e) {
066                            try {
067                                logger.error(e.getMessage(), e);
068                                channel.sendResponse(new XContentThrowableRestResponse(request, e));
069                            } catch (IOException e1) {
070                                logger.error("Failed to send failure response", e1);
071                            }
072                        }
073                    });
074        }
075    }
