055        public static String getFileFunction(String file) {
056            if (file == null || file.trim().length() == 0) {
057                return file;
058            }
059            String res = null;
060            file = file.trim();
061            int idx = file.indexOf('.');
062            String suffix = null;
063            if (idx != -1) {
064                suffix = file.substring(idx + 1);
065            }
066            if (suffix == null) {
067                if (file.startsWith("segments_")) {
068                    return knownExtensions.get(IndexFileNames.SEGMENTS);
069                }
070            } else {
071                res = knownExtensions.get(suffix);
072                if (res != null) {
073                    return res;
074                }
075                // perhaps per-field norms?
076                if (suffix.length() == 2) {
077                    res = knownExtensions.get(suffix.substring(0, 1));
078                }
079            }
080            return res;
081        }

083        public static int getIndexFormat(final Directory dir) throws CorruptIndexException, IOException {
084            SegmentInfos.FindSegmentsFile fsf = new SegmentInfos.FindSegmentsFile(dir) {
085    
086                @Override
087                protected Object doBody(String segmentsFile) throws CorruptIndexException,
088                        IOException {
089                    Integer indexFormat;
090                    IndexInput in = null;
091                    try {
092                        in = dir.openInput(segmentsFile);
093                        indexFormat = new Integer(in.readInt());
094                    } finally {
095                        in.close();
096                    }
097                    return indexFormat;
098                }
099            };
100            Integer indexFormat = (Integer) fsf.run();
101            return indexFormat.intValue();
102        }

167        public static boolean preferCompoundFormat(Directory dir) throws Exception {
168            SegmentInfos infos = new SegmentInfos();
169            infos.read(dir);
170            int compound = 0, nonCompound = 0;
171            for (int i = 0; i < infos.size(); i++) {
172                if (((SegmentInfo) infos.info(i)).getUseCompoundFile()) {
173                    compound++;
174                } else {
175                    nonCompound++;
176                }
177            }
178            return compound > nonCompound;
179        }

181        public static void deletePendingFiles(Directory dir, IndexDeletionPolicy policy) throws Exception {
182            SegmentInfos infos = new SegmentInfos();
183            infos.read(dir);
184            IndexFileDeleter deleter = new IndexFileDeleter(dir, policy, infos, infoStream, null);
185            deleter.close();
186        }

188        public static List<String> getDeletableFiles(Directory dir) throws Exception {
189            SegmentInfos infos = new SegmentInfos();
190            infos.read(dir);
191            IndexFileDeleter deleter = new IndexFileDeleter(dir, new KeepAllIndexDeletionPolicy(), infos, infoStream, null);
192            return (List<String>) deletable.get(deleter);
193        }
194    
195        public static List<String> getIndexFiles(Directory dir) throws Exception {
196            SegmentInfos infos = new SegmentInfos();
197            infos.read(dir);
198            ArrayList<String> names = new ArrayList();
199            for (int i = 0; i < infos.size(); i++) {
200                SegmentInfo info = (SegmentInfo) infos.info(i);
201                names.addAll(info.files());
202                names.add(info.getDelFileName());
203            }
204            names.add(infos.getSegmentsFileName());
205            names.add(IndexFileNames.SEGMENTS_GEN);
206            return names;
207        }

209        public static boolean hasChanges(IndexReader ir) {
210            if (ir == null) {
211                return false;
212            }
213            try {
214                return hasChanges.getBoolean(ir);
215            } catch (IllegalArgumentException  e) {
216                return false;
217            } catch (IllegalAccessException e) {
218                return false;
219            }
220        }
